	// получить следующий объект, для расчета в потоках
	private T getNextItem(List<T> lstItem) {
		Iterator<T> itr = lstItem.iterator();
		T item = null;
		if (itr.hasNext()) {
			item  = itr.next();
			itr.remove();
		}

		return item;
	}
